### Solution 1 Sliding window

<figure><img src="../.gitbook/assets/image (2) (3).png" alt=""><figcaption></figcaption></figure>

```
class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        int maxLen=0;
        int seen[256]={0};
        int l=0;
        int r=0;
        while (r<s.size()){
            seen[s[r]]+=1;
            while (seen[s[r]]>1){
                seen[s[l]]--;
                l++;
            }
            if (r-l+1>maxLen){
                maxLen=r-l+1;
            }
            r++;
        }
        return maxLen;
    }
};
```
